
对象系统，也称值系统或对象表示方法，需要为eval函数返回的内容添加一个定义，也就是需要一个系统。用来表示AST的值或者内存中对AST求值时生成的值。

## 对象系统的基础
增加整数、布尔值、空值的对象
```go
// object.go
package object

//各种对象

import "fmt"

//对象类型
type ObjectType string

const (
	INTEGER_OBJ = "INTEGER"
	BOOLEAN_OBJ = "BOOLEAN"
	NULL_OBJ    = "NULL"
)

//对象接口
type Object interface {
	Type() ObjectType //对象类型
	Inspect() string  //辅助调试用
}

//整数
type Integer struct {
	Value int64
}

func (i *Integer) Inspect() string  { return fmt.Sprintf("%d", i.Value) }
func (i *Integer) Type() ObjectType { return INTEGER_OBJ }

//布尔型
type Boolean struct {
	Value bool
}

func (b *Boolean) Type() ObjectType { return BOOLEAN_OBJ }
func (b *Boolean) Inspect() string  { return fmt.Sprintf("%d", b.Value) }

//空值
type Null struct{}

func (n *Null) Type() ObjectType { return NULL_OBJ }
func (n *Null) Inspect() string  { return "null" }
```
## 整数字面量
新建eval函数处理不同类型的node
```go
//evaluator.go
package evaluator

//求值器

import (
	"monkey_Interpreter/ast"
	"monkey_Interpreter/object"
)

func Eval(node ast.Node) object.Object {
	switch node := node.(type) {

	//语句
	case *ast.Program:
		return evalStatements(node.Statements)

	//表达式语句
	case *ast.ExpressionStatement:
		return Eval(node.Expression)

	//整数字面量
	case *ast.IntegerLiteral:
		return &object.Integer{Value: node.Value}
	}

	return nil
}
```
处理语句
```go
//evaluator.go
func evalStatements(stmts []ast.Statement) object.Object {
	var result object.Object

	for _, statement := range stmts {
		result = Eval(statement)//还是递归调用
	}

	return result
}
```
编写测试函数
```go
//evaluator_test.go
package evaluator

import (
	"monkey_Interpreter/lexer"
	"monkey_Interpreter/object"
	"monkey_Interpreter/parser"
	"testing"
)

// 测试表达式求值
func TestEvalIntegerExpression(t *testing.T) {
	tests := []struct {
		input    string
		expected int64
	}{
		{"5", 5},
		{"10", 10},
	}

	for _, tt := range tests {
		evaluated := testEval(tt.input)
		testIntegerObject(t, evaluated, tt.expected)
	}
}

func testEval(input string) object.Object {
	l := lexer.New(input)
	p := parser.New(l)
	program := p.ParseProgram()

	return Eval(program)
}

func testIntegerObject(t *testing.T, obj object.Object, expected int64) bool {
	result, ok := obj.(*object.Integer)
	if !ok {
		t.Errorf("object is not Integer. got=%T(%+v)", obj, obj)
		return false
	}

	if result.Value != expected {
		t.Errorf("object has wrong value. got=%d, want=%d", result.Value, expected)
		return false
	}
	return true
}
```
