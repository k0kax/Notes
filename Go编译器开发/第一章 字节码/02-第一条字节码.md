第一条字节码，编译`1+2`。
我们的虚拟机采用的是栈式，即先把操作数operands1和2压入栈中，再告诉虚拟机，要让二者相加。即需要两种操作指令Opcode，即压栈和将已存在的栈元素相加。

先定义一个code文件来承载字节码。
```go
// Monkey_Compiler/code/code.go
package code

import (
	"encoding/binary"
	"fmt"
)

type Instructions []byte //字节集合

type Opcode byte //操作码1字节

const (
	OpConstant Opcode = iota //操作码定义
)

// 辅助函数--调试用
type Definition struct {
	Name          string //操作码名称
	OperandWidths []int  //每个操作数所占字节数
}

var definitions = map[Opcode]*Definition{ //全局映射表：记录每个操作码的名称和操作数字节宽度
	OpConstant: {"OpConstant", []int{2}},
}

// Lookup 根据操作码字节值查找对应的元信息
// 用于调试、反汇编字节码时，知道操作码的名称和操作数结构
func Lookup(op byte) (*Definition, error) {
	def, ok := definitions[Opcode(op)]
	if !ok {
		return nil, fmt.Errorf("opcpde %d undefined", op)
	}

	return def, nil
}

// 根据操作码op和操作数operands生成完整的字节码指令
func Make(op Opcode, operands ...int) []byte {
	//查找当前操作码的元信息（名称、操作数宽度）
	def, ok := definitions[op]
	if !ok {
		return []byte{}
	}

	//指令总长度
	instructionLen := 1 //操作码固定1字节
	for _, w := range def.OperandWidths {
		instructionLen += w //加上所有操作数的字节数
	}

	instruction := make([]byte, instructionLen) //存储最终指令 字节码
	instruction[0] = byte(op)                   //第一个字节写入操作码

	offset := 1 //操作数的起始偏移（跳过操作码）
	for i, o := range operands {
		width := def.OperandWidths[i] // 当前操作数的字节宽度
		switch width {
		case 2: // 操作数占2字节：用大端序写入（网络字节序，高位在前）
			binary.BigEndian.PutUint16(instruction[offset:], uint16(o)) //PutUint16：把16位整数o转换成2字节，写入instruction[offset:]的位置
		}
		offset += width
	}

	return instruction//返回字节码
}

func (ins Instructions) String() string {
	return ""
}

```

测试代码：
```go
// Monkey_Compiler/code/code_test.go
package code

import (
	"encoding/binary"
	"testing"
)

func TestMake(t *testing.T) {
	tests := []struct {
		op       Opcode //操作码
		operands []int  //操作数
		expected []byte //期待的字节码
	}{
		{OpConstant, []int{65534}, []byte{byte(OpConstant), 255, 254}}, //OpConstant操作码 + 操作数65534 → 期望生成 [OpConstant, 255, 254]
	}

	for _, tt := range tests {
		instruction := Make(tt.op, tt.operands...)

		if len(instruction) != len(tt.expected) {
			t.Errorf("instruction has wrong length. want=%d,got=%d", len(tt.expected), len(instruction))
		}

		for i, b := range tt.expected {
			if instruction[i] != tt.expected[i] {
				t.Errorf("wrong byte at pos %d. want=%d,got=%d", i, b, instruction[i])
			}
		}
	}
}

func TestReadOperands(t *testing.T) {
	tests := []struct {
		op        Opcode
		operands  []int
		bytesRead int
	}{
		{OpConstant, []int{65535}, 2},
	}

	for _, tt := range tests {
		instruction := Make(tt.op, tt.operands...)

		def, err := Lookup(byte(tt.op))
		if err != nil {
			t.Fatalf("definition not found: %q\n", err)
		}

		operandsRead, n := ReadOperands(def, instruction[1:])
		if n != tt.bytesRead {
			t.Fatalf("n wrong. want=%d,got=%d", tt.bytesRead, n)
		}

		for i, want := range tt.operands {
			if operandsRead[i] != want {
				t.Errorf("operand wrong. want=%d,got=%d", want, operandsRead[i])
			}
		}
	}
}

func ReadOperands(def *Definition, ins Instructions) ([]int, int) {
	operands := make([]int, len(def.OperandWidths))
	offset := 0

	for i, width := range def.OperandWidths {
		switch width {
		case 2:
			operands[i] = int(ReadUint16(ins[offset:]))
		}

		offset += width
	}
	return operands, offset
}

func ReadUint16(ins Instructions) uint16 {
	return binary.BigEndian.Uint16(ins)
}
```
